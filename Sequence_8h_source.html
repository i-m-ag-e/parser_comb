302template <ParseRule P, ParseRule Op>
303 requires(detail::Chainable<typename P::ValueType, typename Op::ValueType>)
304struct Chain {
305 using ValueType = P::ValueType;
306
307 template <typename P1, typename Op1>
308 explicit constexpr Chain(P1&& p, Op1&& op)
309 : parser(std::forward<P1>(p)), op_parser(std::forward<Op1>(op)) {}
310
311 constexpr auto parse(std::string_view input) const
312 -> ParseResult<ValueType> {
313 auto res = parser.parse(input);
314 if (!res)
315 return std::nullopt;
316
317 auto curr = std::move(res->first);
318 input = res->second;
319
320 while (1) {
321 auto op_res = op_parser.parse(input);
322 if (!op_res)
323 break;
324
325 auto next_res = parser.parse(op_res->second);
326 if (!next_res)
327 break;
328
329 curr = op_res->first(std::move(curr), std::move(next_res->first));
330 input = next_res->second;
331 }
332
333 return std::pair{curr, input};
334 }
335
336 private:
337 P parser;
338 Op op_parser;
339};
